# Para guardar lo obtenido, es una opcin usar un sistema de referencia de coordenadas que est proyectado, 
# de forma que se puedan medir distancias y dems (Antes estaba en geogrfico, que es como un globo terrqueo, en proyectado se aplana)
# Se encontr uno adecuado aqu: https://epsg.io/?q=congo&page=2
# Entonces, se cambia para los tres mapas que se tiene

CongoDR=countries[countries.COUNTRY=='Democratic Republic of the Congo']
CongoDRC_4282=CongoDR.to_crs(4282)

cities_DRC=cities[cities.COUNTRY=='DR Congo']
citiesDRC_4282=cities_DRC.to_crs(CongoDR_4282.crs)

rivers_DRC=rivers.to_crs(CongoDR_4282.crs)
riversDRC_4282=gpd.clip(gdf=rivers_4282,mask=CongoDR_4282)
